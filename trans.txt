przycisk = document.querySelectorAll("button");
wynik = document.querySelector("#wynik");
przycisk.forEach(button => {
    button.addEventListener("click", () =>{
            button.textContent = button.textContent === "0" ? "1" : "0";
            const dwa = Array.from(przycisk).map(button => button.textContent).join("");
            const dziesietny = parseInt(dwa, 2);
            wynik.textContent = (" "+dziesietny);
    })    
});
